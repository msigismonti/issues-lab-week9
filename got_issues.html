Step 2: Exploratory visualization

ggplot(data = storms, aes(x = category, y = pressure)) +
  geom_jitter(width = 0.1, 
              alpha = 0.5,
              aes(color = category),
              show.legend = FALSE) +
  labs(x = "Storm Category", 
       y = "Air pressure at storm center (millibars)"
       ) +
  theme_void()
